58 log.info("开始初始化测试用户...");
59
60 // 使用硬编码的默认值代替AuthSystemProperties
61 boolean skipExisting = false;
62 boolean isTestMode = true;
63 String userPrefix = "test_";
64
65 // 检查是否已初始化
66 if (skipExisting && userRepository.count() > 0) {
67 log.info("用户数据已存在，跳过初始化");
68 return;
69 } else if (userRepository.count() > 0 && isTestMode) {
70 // 在测试模式下，可以清除测试用户
71 log.info("测试用户数据已存在，将重新初始化");
72 userRepository.findAll().stream()
73 .filter(user -> user.getUsername().startsWith(userPrefix))
74 .forEach(user -> userRepository.delete(user));
75 }
76
77 try {
78 // 1. 创建测试用户
79 Map<String, User> users = createTestUsers();
80
81 // 2. 加载角色数据
82 Map<String, Role> roles = loadRoles();
83
84 // 3. 加载组织数据
85 Map<String, Long> organizations = loadOrganizations();
86
87 // 4. 建立用户与角色的关联
88 createUserRoleAssociations(users, roles);
89
90 // 5. 建立用户与组织角色的关联
91 createUserOrganizationRoleAssociations(users, roles, organizations);
92

94
95 } catch (Exception e) {
96 log.error("测试用户初始化失败: {}", e.getMessage(), e);
97 throw new RuntimeException("测试用户初始化失败", e);
98 }
