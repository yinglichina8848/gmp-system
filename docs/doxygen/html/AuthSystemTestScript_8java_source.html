135 private Authentication testLogin(String username, String password, boolean expectedSuccess) {
136 try {
137 Authentication auth = authenticationManager.authenticate(
138 new UsernamePasswordAuthenticationToken(username, password));
139
140 if (!expectedSuccess) {
141 throw new AssertionError("登录成功但预期失败: " + username);
142 }
143
144 SecurityContextHolder.getContext().setAuthentication(auth);
145
146 Set<String> authorities = auth.getAuthorities().stream()
147 .map(GrantedAuthority::getAuthority)
148 .collect(Collectors.toSet());
149
150 log.info("用户 {} 登录成功，权限数量: {}", username, authorities.size());
151 return auth;
152
153 } catch (AuthenticationException e) {
154 if (expectedSuccess) {
155 throw new AssertionError("登录失败但预期成功: " + username + ", 原因: " + e.getMessage());
156 }
157
158 log.info("用户 {} 登录失败，预期结果，原因: {}", username, e.getMessage());
159 return null;
160 } catch (Exception e) {
161 log.warn("用户 {} 登录测试时出现异常: {}", username, e.getMessage());
162
163 if (expectedSuccess) {
164 throw e;
165 }
166
167 return null;
168 }
169 }
