102 public Map<String, Object> generateTrendReport(int hours) {
103 Map<String, Object> report = new LinkedHashMap<>();
104 report.put("reportTime", LocalDateTime.now().format(DateTimeFormatter.ISO_DATE_TIME));
105 report.put("reportType", "TREND_ANALYSIS");
106 report.put("timeRange", hours + " hours");
107
108 // 生成趋势数据
109 Map<String, List<Map<String, Object>>> trends = new LinkedHashMap<>();
110
111 // 按时间排序的快照列表
112 List<PerformanceSnapshot> snapshots = new ArrayList<>(historicalData.getOrDefault("SYSTEM", Collections.emptyList()));
113 snapshots.sort(Comparator.comparing(s -> s.timestamp));
114
115 // 过滤最近N小时的数据
116 LocalDateTime cutoffTime = LocalDateTime.now().minusHours(hours);
117 List<PerformanceSnapshot> recentSnapshots = snapshots.stream()
118 .filter(s -> s.timestamp.isAfter(cutoffTime))
119 .collect(java.util.stream.Collectors.toList());
120
121 // 准备时间序列数据
122 List<Map<String, Object>> timeSeries = new ArrayList<>();
123 for (PerformanceSnapshot snapshot : recentSnapshots) {
124 Map<String, Object> dataPoint = new LinkedHashMap<>();
125 dataPoint.put("timestamp", snapshot.timestamp.format(DateTimeFormatter.ISO_DATE_TIME));
126 dataPoint.put("successRate", snapshot.successRate);
127 dataPoint.put("avgResponseTime", snapshot.avgResponseTime);
128 dataPoint.put("totalCalls", snapshot.totalCalls);
129 timeSeries.add(dataPoint);
130 }
131
132 trends.put("system", timeSeries);
133 report.put("trends", trends);
134
135 // 计算趋势变化
136 if (!recentSnapshots.isEmpty()) {
137 PerformanceSnapshot first = recentSnapshots.get(0);
138 PerformanceSnapshot last = recentSnapshots.get(recentSnapshots.size() - 1);
139
140 Map<String, Object> trendChanges = new LinkedHashMap<>();
141 trendChanges.put("successRateChange", String.format("%.2f%%", last.successRate - first.successRate));
142 trendChanges.put("responseTimeChange", String.format("%.2f%%",
143 ((last.avgResponseTime - first.avgResponseTime) / first.avgResponseTime) * 100));
144 trendChanges.put("callVolumeChange", String.format("%.2f%%",
145 ((last.totalCalls - first.totalCalls) / (double)first.totalCalls) * 100));
146
147 report.put("trendChanges", trendChanges);
148 }
149
150 return report;
151 }

156 private List<String> generateRecommendations(McpMonitoringService.SystemMetrics metrics) {
157 List<String> recommendations = new ArrayList<>();
158
159 // 基于成功率的建议
160 if (metrics.getSuccessRate() < 85) {
161 recommendations.add("警告: MCP调用成功率低于85%，建议检查外部系统连接和错误日志");
162 } else if (metrics.getSuccessRate() < 95) {
163 recommendations.add("提示: MCP调用成功率低于95%，可以考虑优化错误处理和重试策略");
164 }
165
166 // 基于响应时间的建议
167 if (metrics.getAverageDuration() > 500) {
168 recommendations.add("警告: 平均响应时间超过500ms，建议检查网络延迟和外部系统性能");
169 } else if (metrics.getAverageDuration() > 200) {
170 recommendations.add("提示: 平均响应时间超过200ms，考虑增加缓存或优化调用参数");
171 }
172
173 // 基于并发调用的建议
174 if (metrics.getActiveCalls() > 100) {
175 recommendations.add("提示: 当前活跃调用数较高，考虑增加线程池容量");
176 }
177
178 // 检查是否有监控数据
179 if (metrics.getTotalCalls() == 0) {
180 recommendations.add("注意: 系统尚未记录任何MCP调用，无法生成详细性能分析");
181 }
182
183 return recommendations;
184 }

190 public void savePerformanceSnapshot() {
191 McpMonitoringService.SystemMetrics metrics = monitoringService.getSystemMetrics();
192
193 // 仅在有数据时保存快照
194 if (metrics.getTotalCalls() > 0) {
195 PerformanceSnapshot snapshot = new PerformanceSnapshot(
196 LocalDateTime.now(),
197 metrics.getSuccessRate(),
198 metrics.getAverageDuration(),
199 metrics.getTotalCalls(),
200 metrics.getFailedCalls()
201 );
202
203 // 保存系统级快照
204 historicalData.computeIfAbsent("SYSTEM", k -> new ArrayList<>())
205 .add(snapshot);
206
207 // 保留最近24小时的数据（大约288个快照）
208 cleanOldSnapshots("SYSTEM", 288);
209 }
210 }

215 private void cleanOldSnapshots(String key, int maxSnapshots) {
216 List<PerformanceSnapshot> snapshots = historicalData.get(key);
217 if (snapshots != null && snapshots.size() > maxSnapshots) {
218 // 移除最旧的数据，保留最新的N个快照
219 snapshots.subList(0, snapshots.size() - maxSnapshots).clear();
220 }
