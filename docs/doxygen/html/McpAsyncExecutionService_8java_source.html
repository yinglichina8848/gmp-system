52 public <T, R> List<R> executeBatchAsync(List<T> items, Function<T, R> processor, String operationName) {
53 if (items == null || items.isEmpty()) {
54 return Collections.emptyList();
55 }
56
57 int batchSize = performanceConfig.getBatch().getBatchSize();
58 int maxConcurrentTasks = performanceConfig.getThreadPool().getMaxThreads();
59
60 // 计算批次数
61 int totalBatches = (int) Math.ceil((double) items.size() / batchSize);
62 List<CompletableFuture<List<R>>> batchFutures = new ArrayList<>();
63
64 // 创建线程池控制并发
65 ExecutorService executorService = new ThreadPoolExecutor(
66 Math.min(maxConcurrentTasks, totalBatches),
67 Math.min(maxConcurrentTasks, totalBatches),
68 0L, TimeUnit.MILLISECONDS,
69 new LinkedBlockingQueue<>(totalBatches)
70 );
71
72 try {
73 // 分批处理
74 for (int i = 0; i < totalBatches; i++) {
75 final int batchIndex = i;
76 CompletableFuture<List<R>> batchFuture = CompletableFuture.supplyAsync(() -> {
77 int start = batchIndex * batchSize;
78 int end = Math.min(start + batchSize, items.size());
79 List<T> batchItems = items.subList(start, end);
80
81 return batchItems.stream()
82 .map(item -> {
83 McpMonitoringService.McpTimer timer = monitoringService.startTimer(operationName);
84 try {
85 R result = processor.apply(item);
86 timer.complete(true);
87 return result;
88 } catch (Exception e) {
89 timer.complete(false);
90 // 记录异常但继续处理其他项目
91 return null;
92 }
93 })
94 .filter(Objects::nonNull)
95 .collect(Collectors.toList());
96 }, executorService);
97
98 batchFutures.add(batchFuture);
99 }
100
101 // 等待所有批次完成
102 CompletableFuture<Void> allOf = CompletableFuture.allOf(
103 batchFutures.toArray(new CompletableFuture[0])
104 );
105
106 // 合并所有结果
107 return allOf.thenApply(v ->
108 batchFutures.stream()
109 .flatMap(future -> future.join().stream())
110 .collect(Collectors.toList())
111 ).join();
112 } finally {
113 executorService.shutdown();
114 }
115 }

120 public <T> Map<String, T> executeParallelAsync(Map<String, Supplier<T>> tasks) {
121 if (tasks == null || tasks.isEmpty()) {
122 return Collections.emptyMap();
123 }
124
125 Map<String, CompletableFuture<T>> futures = new HashMap<>();
126
127 // 为每个任务创建一个CompletableFuture
128 for (Map.Entry<String, Supplier<T>> entry : tasks.entrySet()) {
129 String taskName = entry.getKey();
130 Supplier<T> task = entry.getValue();
131
132 CompletableFuture<T> future = executeAsync(task, taskName);
133 futures.put(taskName, future);
134 }
135
136 // 等待所有任务完成并收集结果
137 Map<String, T> results = new HashMap<>();
138 for (Map.Entry<String, CompletableFuture<T>> entry : futures.entrySet()) {
139 try {
140 T result = entry.getValue().get();
141 results.put(entry.getKey(), result);
142 } catch (Exception e) {
143 // 记录异常但继续处理其他任务
144 results.put(entry.getKey(), null);
145 }
146 }
147
148 return results;
