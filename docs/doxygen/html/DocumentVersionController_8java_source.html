242 private <T> T getFieldValue(Object object, String fieldName) {
243 if (object == null || fieldName == null || fieldName.isEmpty()) {
244 return null;
245 }
246
247 Class<?> clazz = object.getClass();
248 while (clazz != null) {
249 try {
250 Field field = findField(clazz, fieldName);
251 if (field != null) {
252 field.setAccessible(true);
253 return (T) field.get(object);
254 }
255 } catch (Exception e) {
256 // 忽略异常，继续查找父类
257 }
258 clazz = clazz.getSuperclass();
259 }
260 return null;
261 }
