170 @Override

172 public User save(User user) {
173 // 检查用户名、邮箱、员工编号是否已存在
174 if (user.getId() == null) {
175 // 新增用户时检查
176 if (existsByUsername(user.getUsername())) {
177 throw new IllegalArgumentException("Username already exists: " + user.getUsername());
178 }
179 if (existsByEmail(user.getEmail())) {
180 throw new IllegalArgumentException("Email already exists: " + user.getEmail());
181 }
182 if (existsByEmployeeNumber(user.getEmployeeNumber())) {
183 throw new IllegalArgumentException("Employee number already exists: " + user.getEmployeeNumber());
184 }
185 // 加密密码
186 user.setPassword(passwordEncoder.encode(user.getPassword()));
187 } else {
188 // 更新用户时，如果密码被修改则加密
189 User existingUser = getById(user.getId());
190 if (!existingUser.getPassword().equals(user.getPassword())) {
191 user.setPassword(passwordEncoder.encode(user.getPassword()));
192 }
193 }
194 return super.save(user);
