59 @Override
60 public String uploadFile(MultipartFile file, String bucketName, String prefix) throws Exception {
61 // 如果没有指定桶名，使用默认桶
62 if (bucketName == null || bucketName.isEmpty()) {
63 bucketName = defaultBucket;
64 }
65
66 // 确保桶存在
67 if (!bucketExists(bucketName)) {
68 createBucket(bucketName);
69 }
70
71 // 生成唯一的文件名
72 String originalFilename = file.getOriginalFilename();
73 String extension = FilenameUtils.getExtension(originalFilename);
74 String timestamp = LocalDateTime.now().format(DateTimeFormatter.ofPattern("yyyyMMdd_HHmmss_SSS"));
75 String uuid = UUID.randomUUID().toString().replace("-", "");
76 String filename = timestamp + "_" + uuid + (extension.isEmpty() ? "" : "." + extension);
77
78 // 构建文件路径
79 String filePath = prefix != null && !prefix.isEmpty()
80 ? prefix.endsWith("/") ? prefix + filename : prefix + "/" + filename
81 : filename;
82
83 // 上传文件
84 minioClient.putObject(
85 PutObjectArgs.builder()
86 .bucket(bucketName)
87 .object(filePath)
88 .stream(file.getInputStream(), file.getSize(), ObjectWriteArgs.MIN_MULTIPART_SIZE)
89 .contentType(file.getContentType())
90 .build());
91
92 return filePath;
93 }

121 @Override
122 public void batchDeleteFiles(String bucketName, List<String> filePaths) throws Exception {
123 if (bucketName == null || bucketName.isEmpty()) {
124 bucketName = defaultBucket;
125 }
126
127 List<DeleteObject> deleteObjects = filePaths.stream()
128 .map(filePath -> new DeleteObject(filePath))
129 .collect(Collectors.toList());
130
131 Iterable<Result<DeleteError>> results = minioClient.removeObjects(
132 RemoveObjectsArgs.builder()
133 .bucket(bucketName)
134 .objects(deleteObjects)
135 .build());
136
137 // 检查是否有删除失败的文件
138 List<String> failedDeletions = new ArrayList<>();
139 for (Result<DeleteError> result : results) {
140 try {
141 DeleteError error = result.get();
142 failedDeletions.add(error.objectName() + ": " + error.message());
143 } catch (Exception e) {
144 failedDeletions.add("Unknown error: " + e.getMessage());
145 }
146 }
147
148 if (!failedDeletions.isEmpty()) {
149 throw new Exception("Some files failed to delete: " + String.join(", ", failedDeletions));
150 }
151 }

389 contentType = "application/vnd.ms-powerpoint";
390 } else if (filePath.endsWith(".jpg") || filePath.endsWith(".jpeg")) {
391 contentType = "image/jpeg";

393 contentType = "image/png";

444 if (bucketName == null || bucketName.isEmpty()) {

446 }
447
448 // 检查桶是否存在

452
453 // 统计桶中的对象数量和总大小
454 long totalSize = 0;
455 int objectCount = 0;
456
457 Iterable<Result<Item>> results = minioClient.listObjects(
458 ListObjectsArgs.builder()
459 .bucket(bucketName)
460 .recursive(true)
461 .build());
462
463 for (Result<Item> result : results) {
464 Item item = result.get();
465 if (!item.isDir()) {
466 totalSize += item.size();
467 objectCount++;
468 }
469 }
470
471 Map<String, Long> storageInfo = new HashMap<>();
472 storageInfo.put("objectCount", (long) objectCount);
473 storageInfo.put("totalSize", totalSize);
474
475 return storageInfo;
