53 @Override
54 @Transactional
55 public CommonFileDTO uploadFile(MultipartFile file, String module, Map<String, Object> metadata) throws Exception {
56 // 验证文件
57 if (file == null || file.isEmpty()) {
58 throw new IllegalArgumentException("文件不能为空");
59 }
60
61 // 验证模块名称
62 if (module == null || module.trim().isEmpty()) {
63 throw new IllegalArgumentException("模块名称不能为空");
64 }
65
66 // 计算文件校验和
67 String checksum = calculateChecksum(file);
68
69 // 检查文件是否已存在
70 Optional<CommonFile> existingFile = commonFileRepository.findByChecksum(checksum);
71 if (existingFile.isPresent()) {
72 // 文件已存在，可以选择返回已存在的文件信息或抛出异常
73 return convertToDTO(existingFile.get());
74 }
75
76 // 生成文件路径
77 String fileName = file.getOriginalFilename();
78 String filePath = generateFilePath(module, fileName);
79
80 // 创建文件实体
81 CommonFile commonFile = new CommonFile();
82 commonFile.setFileName(fileName);
83 commonFile.setFileType(file.getContentType());
84 commonFile.setFileSize(file.getSize());
85 commonFile.setFilePath(filePath);
86 commonFile.setChecksum(checksum);
87 commonFile.setBucketName(commonFilesBucket);
88 commonFile.setModule(module);
89 commonFile.setCreatedBy("system"); // 实际应用中应该从认证信息获取
90
91 // 设置元数据
92 if (metadata != null && !metadata.isEmpty()) {
93 commonFile.setMetadata(objectMapper.valueToTree(metadata));
94 }
95
96 // 保存文件信息到数据库
97 commonFile = commonFileRepository.save(commonFile);
98
99 try {
100 // 上传文件到存储服务
101 fileStorageService.uploadFile(file, commonFilesBucket, filePath);
102
103 // 转换为DTO并返回
104 return convertToDTO(commonFile);
105 } catch (Exception e) {
106 // 如果文件上传失败，删除数据库记录
107 commonFileRepository.delete(commonFile);
108 throw e;
109 }
110 }

187 @Override
188 public Map<String, Object> queryFiles(String module, Map<String, Object> filters, int page, int size) {
189 Pageable pageable = PageRequest.of(page - 1, size);
190 Page<CommonFile> filePage;
191
192 // 根据过滤条件构建查询
193 if (filters != null && filters.containsKey("keyword")) {
194 String keyword = (String) filters.get("keyword");
195 filePage = commonFileRepository.searchByFileName(keyword, pageable);
196 } else if (module != null && !module.isEmpty()) {
197 if (filters != null && filters.containsKey("status")) {
198 String status = (String) filters.get("status");
199 filePage = commonFileRepository.findByModuleAndStatus(module, status, pageable);
200 } else {
201 filePage = commonFileRepository.findByModule(module, pageable);
202 }
203 } else {
204 filePage = commonFileRepository.findByStatusOrderByCreatedAtDesc("ACTIVE", pageable);
205 }
206
207 List<CommonFileDTO> fileDTOs = filePage.getContent().stream()
208 .map(this::convertToDTO)
209 .collect(Collectors.toList());
210
211 Map<String, Object> result = new HashMap<>();
212 result.put("content", fileDTOs);
213 result.put("totalElements", filePage.getTotalElements());
214 result.put("totalPages", filePage.getTotalPages());
215 result.put("pageNumber", filePage.getNumber() + 1);
216 result.put("pageSize", filePage.getSize());
217
218 return result;
219 }

272 private String calculateChecksum(MultipartFile file) throws Exception {
273 MessageDigest digest = MessageDigest.getInstance("MD5");
274 byte[] bytes = file.getBytes();
275 digest.update(bytes);
276 byte[] hash = digest.digest();
277
278 StringBuilder hexString = new StringBuilder();
279 for (byte b : hash) {
280 hexString.append(String.format("%02x", b));
281 }
282 return hexString.toString();
283 }

317 private String formatFileSize(Long size) {
318 if (size == null) return "0 B";
319
320 if (size < 1024) return size + " B";
321 else if (size < 1024 * 1024) return String.format("%.2f KB", size / 1024.0);
322 else if (size < 1024 * 1024 * 1024) return String.format("%.2f MB", size / (1024.0 * 1024));
323 else return String.format("%.2f GB", size / (1024.0 * 1024 * 1024));
324 }
