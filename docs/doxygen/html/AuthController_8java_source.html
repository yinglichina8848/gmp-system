187 private String getClientIp(HttpServletRequest request) {
188 String xForwardedFor = request.getHeader("X-Forwarded-For");
189 if (xForwardedFor != null && !xForwardedFor.isEmpty()) {
190 return xForwardedFor.split(",")[0].trim();
191 }
192
193 String xRealIp = request.getHeader("X-Real-IP");
194 if (xRealIp != null && !xRealIp.isEmpty()) {
195 return xRealIp;
196 }
197
198 return request.getRemoteAddr();
199 }

204 private String extractUsernameFromToken(String authHeader) {
205 try {
206 if (authHeader != null && authHeader.startsWith("Bearer ")) {
207 // 提取实际的token部分（去掉Bearer前缀）
208 String token = authHeader.substring(7);
209 // 使用JwtConfig来解析令牌获取用户名
210 return jwtConfig.extractUsername(token);
211 }
212 } catch (Exception e) {
213 log.error("从令牌中提取用户名失败: {}", e.getMessage());
