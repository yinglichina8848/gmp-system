186 private String getClientIp(HttpServletRequest request) {
187 String xForwardedFor = request.getHeader("X-Forwarded-For");
188 if (xForwardedFor != null && !xForwardedFor.isEmpty()) {
189 return xForwardedFor.split(",")[0].trim();
190 }
191
192 String xRealIp = request.getHeader("X-Real-IP");
193 if (xRealIp != null && !xRealIp.isEmpty()) {
194 return xRealIp;
195 }
196
197 return request.getRemoteAddr();
198 }

203 private String extractUsernameFromToken(String authHeader) {
204 try {
205 if (authHeader != null && authHeader.startsWith("Bearer ")) {
206 // 提取实际的token部分（去掉Bearer前缀）
207 String token = authHeader.substring(7);
208 // 使用JwtConfig来解析令牌获取用户名
209 return jwtConfig.extractUsername(token);
210 }
211 } catch (Exception e) {
212 log.error("从令牌中提取用户名失败: {}", e.getMessage());
213 }
214 return "anonymous";
