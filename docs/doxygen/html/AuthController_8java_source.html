257 private String getClientIp(HttpServletRequest request) {
258 String xForwardedFor = request.getHeader("X-Forwarded-For");
259 if (xForwardedFor != null && !xForwardedFor.isEmpty()) {
260 return xForwardedFor.split(",")[0].trim();
261 }
262
263 String xRealIp = request.getHeader("X-Real-IP");
264 if (xRealIp != null && !xRealIp.isEmpty()) {
265 return xRealIp;
266 }
267
268 return request.getRemoteAddr();
269 }

279 private String extractUsernameFromToken(String authHeader) {
280 try {
281 if (authHeader != null && authHeader.startsWith("Bearer ")) {
282 // 提取实际的token部分（去掉Bearer前缀）
283 String token = authHeader.substring(7);
284 // 使用JwtConfig来解析令牌获取用户名
285 return jwtConfig.extractUsername(token);
286 }
287 } catch (Exception e) {
288 log.error("从令牌中提取用户名失败: {}", e.getMessage());
289 }
290 return "anonymous";
291 }
