35 @ExceptionHandler(AuthenticationException.class)
36 @ResponseStatus(HttpStatus.UNAUTHORIZED)
37 @ResponseBody
38 public ResponseEntity<ErrorResponse> handleAuthenticationException(AuthenticationException ex) {
39 logger.error("Authentication error: {}", ex.getMessage());
40
41 String errorMessage = "认证失败";
42 if (ex instanceof BadCredentialsException) {
43 errorMessage = "用户名或密码错误";
44 } else if (ex instanceof DisabledException) {
45 errorMessage = "账号已被禁用";
46 } else if (ex instanceof LockedException) {
47 errorMessage = "账号已被锁定";
48 }
49
50 ErrorResponse errorResponse = new ErrorResponse(
51 HttpStatus.UNAUTHORIZED.value(),
52 "UNAUTHORIZED",
53 errorMessage,
54 "/api/auth"
55 );
56
57 return new ResponseEntity<>(errorResponse, HttpStatus.UNAUTHORIZED);
58 }
