209 private String generateTestReport() {
210 StringBuilder report = new StringBuilder();
211
212 report.append("\n=======================================\n");
213 report.append(" GMP认证测试执行报告 \n");
214 report.append("=======================================\n");
215 report.append("开始时间: ").append(startTime).append("\n");
216 report.append("结束时间: ").append(endTime).append("\n");
217
218 // 计算总执行时间
219 if (startTime != null && endTime != null) {
220 long executionTime = startTime.until(endTime, java.time.temporal.ChronoUnit.MILLIS);
221 report.append("总执行时间: ").append(executionTime).append("ms\n");
222 }
223
224 report.append("\n");
225
226 // 测试状态明细
227 report.append("测试方法执行状态:\n");
228 Map<String, AuthTestErrorHandler.TestStatus> statuses = errorHandler.getAllTestStatuses();
229
230 for (String testName : statuses.keySet()) {
231 AuthTestErrorHandler.TestStatus status = statuses.get(testName);
232 report.append(" [")
233 .append(status.isSuccess() ? "✓" : "✗")
234 .append("] ")
235 .append(testName)
236 .append(" (时间: ").append(status.getExecutionTime()).append("ms")
237 .append(status.isSuccess() ? "" : ", 错误: " + status.getErrorType() + ": " + status.getErrorMessage())
238 .append(")\n");
239 }
240
241 report.append("\n");
242
243 // 错误统计
244 report.append("错误统计:\n");
245 Map<AuthTestErrorHandler.ErrorType, Integer> errorStats = errorHandler.getErrorStatistics();
246 boolean hasErrors = false;
247
248 for (Map.Entry<AuthTestErrorHandler.ErrorType, Integer> entry : errorStats.entrySet()) {
249 if (entry.getValue() > 0) {
250 report.append(" " + entry.getKey() + ": " + entry.getValue() + "次\n");
251 hasErrors = true;
252 }
253 }
254
255 if (!hasErrors) {
256 report.append(" 无错误\n");
257 }
258
259 report.append("\n");
260
261 // 总体结果
262 int totalTests = statuses.size();
263 int passedTests = (int) statuses.values().stream().filter(AuthTestErrorHandler.TestStatus::isSuccess).count();
264
265 report.append("测试结果: ").append(passedTests).append("/").append(totalTests).append(" 通过")
266 .append(" (").append(Math.round((double) passedTests / totalTests * 100)).append("%)\n");
267
268 if (passedTests == totalTests) {
269 report.append("✅ 所有测试通过！认证系统功能正常。\n");
270 } else {
271 report.append("❌ 部分测试失败，请检查错误信息。\n");
272 }
273
274 report.append("=======================================\n");
275
276 return report.toString();
277 }
