223 public class McpTimer {
224 private final String operationName;
225 private final long startTime;
226
227 public McpTimer(String operationName) {
228 this.operationName = operationName;
229 this.startTime = System.currentTimeMillis();
230 }
231
236 public void complete(boolean success) {
237 long duration = System.currentTimeMillis() - startTime;
238
239 // 记录性能指标
240 performanceMetrics.computeIfAbsent(operationName, k -> Collections.synchronizedList(new ArrayList<>()))
241 .add(duration);
242
243 // 更新系统调用统计
244 SystemCallStats stats = systemCallStats.computeIfAbsent(operationName, k -> new SystemCallStats());
245 stats.incrementCalls();
246 if (success) {
247 stats.incrementSuccesses();
248 } else {
249 stats.incrementFailures();
250 }
251 stats.addResponseTime(duration);
252
253 log.debug("Operation {} completed in {}ms, success: {}", operationName, duration, success);
254 }
