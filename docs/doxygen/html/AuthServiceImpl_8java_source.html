49 @Transactional
50 public LoginResponse login(LoginRequest request, String ipAddress, String userAgent) {
51 String username = request.getUsername();
52 String password = request.getPassword();
53
54 // 查找用户
55 User user = userRepository.findByUsername(username)
56 .orElseThrow(() -> new RuntimeException("用户名或密码错误"));
57
58 // 检查用户状态
59 if (user.getUserStatus() == User.UserStatus.LOCKED) {
60 // 记录失败登录
61 logFailedLoginAttempt(user, "账户已锁定", ipAddress, userAgent);
62 throw new RuntimeException("账户已被锁定，请联系管理员");
63 }
64
65 // 检查密码
66 if (!passwordEncoder.matches(password, user.getPassword())) {
67 // 增加登录尝试次数
68 user.incrementLoginAttempts();
69
70 // 如果尝试次数超过限制，锁定账户
71 if (user.getLoginAttempts() >= 5) {
72 user.lockAccount(15); // 锁定15分钟
73 logFailedLoginAttempt(user, "密码错误，账户已锁定", ipAddress, userAgent);
74 throw new RuntimeException("密码错误次数过多，账户已被锁定");
75 }
76
77 userRepository.save(user);
78 logFailedLoginAttempt(user, "密码错误", ipAddress, userAgent);
79 throw new RuntimeException("用户名或密码错误");
80 }
81
82 // 登录成功，重置登录尝试次数
83 user.resetLoginAttempts();
84 user.setLastLoginTime(LocalDateTime.now());
85 user.setLastLoginIp(ipAddress);
86 userRepository.save(user);
87
88 // 获取用户角色和权限
89 Set<String> roles = new HashSet<>();
90 List<String> permissions = new ArrayList<>();
91
92 // 假设roles现在已经是字符串集合
93 roles.addAll(user.getRoles());
94 // 假设permissions已经是字符串列表
95 permissions.addAll(user.getPermissions());
96 // 注意：这里假设User类已经实现了getPermissions()方法
97
98 // 生成JWT令牌
99 String accessToken = jwtConfig.generateToken(username, roles, permissions);
100 String refreshToken = jwtConfig.generateRefreshToken(username);
101
102 // 记录成功登录
103 logSuccessfulLogin(user, ipAddress, userAgent);
104
105 // 创建登录响应
106 LoginResponse response = new LoginResponse();
107 // 直接设置字段值，不使用setter方法以避免Lombok问题
108 response.accessToken = accessToken;
109 response.refreshToken = refreshToken;
110 response.tokenType = "Bearer";
111 response.expiresIn = jwtConfig.getExpiration() / 1000;
112 response.username = user.getUsername();
113 response.fullName = user.getFullName();
114 response.roles = new ArrayList<>(roles);
115 response.permissions = permissions;
116 response.loginTime = LocalDateTime.now();
117 return response;
118 }
