355 private List<String> listObjects(String bucketName, String prefix) {
356 List<String> objects = new ArrayList<>();
357 try {
358 Iterable<Result<io.minio.messages.Item>> results = minioClient.listObjects(
359 ListObjectsArgs.builder()
360 .bucket(bucketName)
361 .prefix(prefix)
362 .recursive(true)
363 .build()
364 );
365 for (Result<io.minio.messages.Item> result : results) {
366 objects.add(result.get().objectName());
367 }
368 } catch (Exception e) {
369 // 忽略错误
370 }
371 return objects;
372 }
