174 @Transactional
175 public void initialize() {
176 // 简化实现，避免getInitialization()方法错误
177 // 假设始终需要初始化
178 log.info("开始角色和权限初始化");
179
180 try {
181 // 简化实现，避免getInitialization()方法错误
182 boolean skipExisting = false; // 假设不跳过已有数据
183
184 // 检查是否已存在角色数据
185 if (skipExisting && roleRepository.count() > 0) {
186 log.info("角色数据已存在，跳过初始化");
187 return;
188 } else if (roleRepository.count() > 0) {
189 log.info("角色数据已存在，将重新初始化");
190 rolePermissionRepository.deleteAll();
191 permissionRepository.deleteAll();
192 roleRepository.deleteAll();
193 }
194
195 // 1. 创建预设角色
196 Map<String, Role> roles = createPredefinedRoles();
197
198 // 2. 创建权限
199 Map<String, Permission> permissions = createPermissions();
200
201 // 3. 分配权限给角色
202 assignPermissionsToRoles(roles, permissions);
203
204 log.info("角色权限初始化完成，共创建 {} 个角色，{} 个权限，{} 个角色-权限关联",
205 roles.size(), permissions.size(), rolePermissionRepository.count());
206
207 } catch (Exception e) {
208 log.error("角色权限初始化失败: {}", e.getMessage(), e);
209 throw new RuntimeException("角色权限初始化失败", e);
210 }
211 }
