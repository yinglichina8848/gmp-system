146 @Transactional
147 public BatchOperation startBatchOperation(Long id, Map<String, Object> parameters, String operator) {
148 Optional<BatchOperation> optionalOperation = batchOperationRepository.findById(id);
149 if (optionalOperation.isPresent()) {
150 BatchOperation operation = optionalOperation.get();
151
152 // 检查操作状态
153 if (operation.getStatus() != BatchOperation.OperationStatus.PENDING) {
154 throw new IllegalStateException("Cannot start operation, current status: " + operation.getStatus());
155 }
156
157 // 更新操作状态和参数
158 operation.setStatus(BatchOperation.OperationStatus.IN_PROGRESS);
159 try {
160 operation.setParameters(objectMapper.writeValueAsString(parameters));
161 } catch (Exception e) {
162 operation.setParameters("{}");
163 }
164 // 使用重命名后的方法设置Map形式的parameters（兼容测试）
165 operation.setParametersMap(parameters);
166 operation.setStartTime(LocalDateTime.now());
167 operation.setOperator(operator);
168
169 // 生成操作名称 (如果没有)
170 if (operation.getOperationName() == null || operation.getOperationName().isEmpty()) {
171 operation.setOperationName("OP-" + generateOperationNumber());
172 }
173
174 return batchOperationRepository.save(operation);
175 } else {
176 throw new IllegalArgumentException("Batch operation not found: " + id);
