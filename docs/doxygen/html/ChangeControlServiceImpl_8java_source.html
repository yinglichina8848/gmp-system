96 @Override

99 ChangeControl changeControl = getChangeControlById(changeControlId);
100
101 try {
102 // 确保上传目录存在
103 Path uploadPath = Paths.get(UPLOAD_DIR + changeControlId + "/");
104 Files.createDirectories(uploadPath);
105
106 // 生成唯一文件名
107 String originalFilename = file.getOriginalFilename();
108 String extension = originalFilename != null ? originalFilename.substring(originalFilename.lastIndexOf('.')) : "";
109 String fileName = UUID.randomUUID() + extension;
110
111 // 保存文件
112 Path filePath = uploadPath.resolve(fileName);
113 Files.copy(file.getInputStream(), filePath);
114
115 // 创建附件记录
116 ChangeControlAttachment attachment = new ChangeControlAttachment();
117 attachment.setChangeControl(changeControl);
118 attachment.setFileName(originalFilename);
119 attachment.setFilePath(filePath.toString());
120 attachment.setFileSize(file.getSize());
121 attachment.setFileType(file.getContentType());
122 attachment.setDescription(description);
123
124 changeControl.getAttachments().add(attachment);
125 return changeControlRepository.save(changeControl);
126 } catch (IOException e) {
127 throw new RuntimeException("Failed to upload file: " + e.getMessage(), e);
128 }
129 }
