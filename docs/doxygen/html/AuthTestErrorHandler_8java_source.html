60 public <T> T handleTestMethod(String testMethodName, TestMethod<T> method) {
61 TestStatus status = new TestStatus(testMethodName);
62 testStatusMap.put(testMethodName, status);
63
64 log.info("开始执行测试方法: {}", testMethodName);
65 status.setStartTime(LocalDateTime.now());
66
67 int retryCount = 0;
68 while (retryCount <= MAX_RETRY_COUNT) {
69 try {
70 T result = method.execute();
71 status.setSuccess(true);
72 status.setEndTime(LocalDateTime.now());
73 log.info("测试方法 {} 执行成功", testMethodName);
74 return result;
75
76 } catch (Exception e) {
77 ErrorType errorType = classifyError(e);
78 handleException(e, testMethodName, errorType, retryCount);
79
80 // 增加错误计数
81 errorCountMap.put(errorType, errorCountMap.get(errorType) + 1);
82
83 // 记录错误
84 ErrorRecord errorRecord = new ErrorRecord(
85 testMethodName,
86 errorType,
87 e.getMessage(),
88 LocalDateTime.now(),
89 retryCount
90 );
91 errorHistory.add(errorRecord);
92
93 // 更新测试状态
94 status.setSuccess(false);
95 status.setErrorMessage(e.getMessage());
96 status.setErrorType(errorType);
97
98 // 判断是否需要重试
99 if (shouldRetry(errorType, retryCount)) {
100 retryCount++;
101 log.warn("将重试测试方法 {}，重试次数: {}/{}",
102 testMethodName, retryCount, MAX_RETRY_COUNT);
103 try {
104 // 重试前等待一段时间
105 Thread.sleep(calculateRetryDelay(retryCount));
106 } catch (InterruptedException ie) {
107 Thread.currentThread().interrupt();
108 log.error("重试等待被中断", ie);
109 break;
110 }
111 } else {
112 status.setEndTime(LocalDateTime.now());
113 throw new TestExecutionException("测试方法 " + testMethodName + " 执行失败，已达到最大重试次数", e);
114 }
115 }
116 }
117
118 status.setEndTime(LocalDateTime.now());
119 throw new TestExecutionException("测试方法 " + testMethodName + " 执行失败，已达到最大重试次数");
120 }

125 private ErrorType classifyError(Exception e) {
126 if (e instanceof AuthenticationException) {
127 return ErrorType.AUTHENTICATION_ERROR;
128 } else if (e instanceof DataAccessException || e instanceof SQLException) {
129 return ErrorType.DATABASE_ERROR;
130 } else if (e instanceof IllegalStateException || e instanceof IllegalArgumentException) {
131 return ErrorType.VALIDATION_ERROR;
132 } else if (e instanceof NoSuchElementException || e instanceof ClassNotFoundException) {
133 return ErrorType.INITIALIZATION_ERROR;
134 } else if (e instanceof TestExecutionException) {
135 return ((TestExecutionException) e).getErrorType();
136 } else {
137 return ErrorType.OTHER_ERROR;
138 }
139 }

144 private void handleException(Exception e, String testMethodName, ErrorType errorType, int retryCount) {
145 String errorMessage = String.format("测试方法 %s 执行失败 (类型: %s, 重试次数: %d): %s",
146 testMethodName, errorType, retryCount, e.getMessage());
147
148 switch (errorType) {
149 case DATABASE_ERROR:
150 log.error(errorMessage + ", 数据库连接或操作失败", e);
151 // 这里可以添加数据库连接恢复逻辑
152 break;
153 case AUTHENTICATION_ERROR:
154 log.warn(errorMessage + ", 认证失败");
155 break;
156 case INITIALIZATION_ERROR:
157 log.error(errorMessage + ", 初始化资源失败", e);
158 break;
159 case VALIDATION_ERROR:
160 log.error(errorMessage + ", 参数或数据验证失败", e);
161 break;
162 default:
163 log.error(errorMessage, e);
164 }
165 }

170 private boolean shouldRetry(ErrorType errorType, int retryCount) {
171 if (retryCount >= MAX_RETRY_COUNT) {
172 return false;
173 }
174
175 // 只对特定类型的错误进行重试
176 return errorType == ErrorType.DATABASE_ERROR ||
177 errorType == ErrorType.NETWORK_ERROR ||
178 errorType == ErrorType.INITIALIZATION_ERROR;
179 }
180
184 private long calculateRetryDelay(int retryCount) {
185 // 基础延迟时间 1000ms
186 long baseDelay = 1000;
187 // 指数退避，最大不超过 5s
188 return Math.min(baseDelay * (long) Math.pow(2, retryCount), 5000);
189 }

222 public String generateErrorReport() {
223 StringBuilder report = new StringBuilder();
224 report.append("\n=======================================\n");
225 report.append(" GMP认证测试错误报告 \n");
226 report.append("=======================================\n");
227 report.append("生成时间: ").append(LocalDateTime.now()).append("\n\n");
228
229 // 测试状态统计
230 int totalTests = testStatusMap.size();
231 int passedTests = (int) testStatusMap.values().stream().filter(TestStatus::isSuccess).count();
232 int failedTests = totalTests - passedTests;
233
234 report.append("测试统计:\n");
235 report.append(" 总测试数: " + totalTests + "\n");
236 report.append(" 通过测试: " + passedTests + "\n");
237 report.append(" 失败测试: " + failedTests + "\n\n");
238
239 // 错误类型统计
240 report.append("错误类型统计:\n");
241 errorCountMap.forEach((type, count) -> {
242 if (count > 0) {
243 report.append(" " + type + ": " + count + "次\n");
244 }
245 });
246 report.append("\n");
247
248 // 详细错误信息
249 if (!errorHistory.isEmpty()) {
250 report.append("详细错误记录:\n");
251 for (ErrorRecord record : errorHistory) {
252 report.append(" [" + record.getTimestamp() + "] " +
253 record.getTestName() + ": " +
254 record.getErrorType() + " - " +
255 record.getErrorMessage() +
256 " (重试: " + record.getRetryCount() + ")\n");
257 }
258 }
259
260 report.append("=======================================\n");
261 return report.toString();
262 }
