57 @Override
58 public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) throws Exception {
59 String requestUri = request.getRequestURI();
60 String method = request.getMethod();
61
62 log.debug("拦截请求: {} {}", method, requestUri);
63
64 // 检查是否需要排除拦截
65 if (shouldExclude(requestUri)) {
66 log.debug("路径已排除拦截: {}", requestUri);
67 return true;
68 }
69
70 // 从请求头获取令牌
71 String token = extractTokenFromHeader(request);
72 if (token == null) {
73 log.warn("请求缺少令牌: {}", requestUri);
74 sendErrorResponse(response, "MISSING_TOKEN", "缺少访问令牌");
75 return false;
76 }
77
78 // 验证令牌
79 if (!jwtConfig.validateToken(token)) {
80 log.warn("无效的令牌: {}", requestUri);
81 sendErrorResponse(response, "INVALID_TOKEN", "无效的访问令牌");
82 return false;
83 }
84
85 // 从令牌中提取用户名
86 String username = jwtConfig.extractUsername(token);
87 if (username == null) {
88 log.warn("令牌中缺少用户名: {}", requestUri);
89 sendErrorResponse(response, "INVALID_TOKEN", "令牌格式错误");
90 return false;
91 }
92
93 // 确定请求对应的子系统代码
94 String subsystemCode = determineSubsystemCode(requestUri);
95 if (subsystemCode != null) {
96 // 检查用户对子系统的访问权限
97 if (!authService.hasSubsystemAccess(username, subsystemCode)) {
98 log.warn("用户[{}]无权访问子系统[{}]: {}", username, subsystemCode, requestUri);
99 sendErrorResponse(response, "ACCESS_DENIED", "无权访问该子系统");
100 return false;
101 }
102
103 // 对于需要写入权限的操作，检查是否有写入权限
104 if (isWriteOperation(method) && !authService.hasSubsystemWriteAccess(username, subsystemCode)) {
105 log.warn("用户[{}]对子系统[{}]无写入权限: {}", username, subsystemCode, requestUri);
106 sendErrorResponse(response, "WRITE_ACCESS_DENIED", "对子系统无写入权限");
107 return false;
108 }
109
110 log.debug("用户[{}]对子系统[{}]的访问权限验证通过", username, subsystemCode);
111 }
112
113 return true;
114 }

119 private boolean shouldExclude(String requestUri) {
