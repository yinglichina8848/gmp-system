57 // 从令牌中获取所有声明
58 private Claims getAllClaimsFromToken(String token) {
59 try {
60 return Jwts.parserBuilder()
61 .setSigningKey(getSigningKey())
62 .setClock(() -> new Date())
63 .build()
64 .parseClaimsJws(token)
65 .getBody();
66 } catch (SignatureException e) {
67 logger.error("JWT签名验证失败: {}", e.getMessage());

72 } catch (MalformedJwtException e) {

85 private Key getSigningKey() {
86 // 确保密钥长度足够安全（至少256位）
87 byte[] keyBytes = jwtSecret.getBytes();
88 if (keyBytes.length < 32) { // 256位 = 32字节
89 logger.warn("JWT密钥长度不足，建议至少32字节");

91 return Keys.hmacShaKeyFor(keyBytes);

145 // 创建令牌
146 private String doGenerateToken(Map<String, Object> claims, String subject, long expiration) {
147 String tokenId = UUID.randomUUID().toString();
148
149 return Jwts.builder()
150 .setClaims(claims)
151 .setSubject(subject)
152 .setId(tokenId)
153 .setIssuedAt(new Date(System.currentTimeMillis()))
154 .setExpiration(new Date(System.currentTimeMillis() + expiration * 1000))
155 .setIssuer("GMP-Auth-Service")
156 .setNotBefore(new Date(System.currentTimeMillis()))
157 .signWith(getSigningKey(), SignatureAlgorithm.HS512)
158 .compact();
159 }

161 // 验证令牌
162 public Boolean validateToken(String token, UserDetails userDetails) {
163 try {
164 // 检查token和userDetails是否为null
165 if (token == null || userDetails == null) {
166 logger.warn("JWT令牌或用户详情为空，验证失败");
167 return false;
168 }
169
170 final String username = getUsernameFromToken(token);
171 return (username.equals(userDetails.getUsername()) &&
172 !isTokenExpired(token) &&
173 !isTokenRevoked(token));
174 } catch (Exception e) {
175 logger.error("JWT令牌验证异常: {}", e.getMessage());
176 return false;
177 }
178 }

181 // 仅检查令牌是否过期和是否被撤销

201 // 验证签名
202 Jwts.parserBuilder().setSigningKey(getSigningKey()).build().parseClaimsJws(token);
203 return true;
204 } catch (SignatureException e) {
205 logger.error("JWT签名验证失败: {}", e.getMessage());
206 return false;
207 } catch (MalformedJwtException e) {
208 logger.error("无效的JWT令牌: {}", e.getMessage());
209 return false;
210 } catch (ExpiredJwtException e) {
211 logger.error("JWT令牌已过期: {}", e.getMessage());
212 return false;
213 } catch (UnsupportedJwtException e) {
214 logger.error("不支持的JWT令牌: {}", e.getMessage());
215 return false;
216 } catch (IllegalArgumentException e) {
217 logger.error("JWT声明字符串为空: {}", e.getMessage());
218 return false;
219 } catch (Exception e) {
