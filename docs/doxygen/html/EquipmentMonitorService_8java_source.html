224 private void checkAndUpdateStatusForThresholds(EquipmentMonitor equipment) {
225 boolean isError = false;
226 String errorReason = "";
227
228 // 检查温度阈值
229 if (equipment.getTemperatureThreshold() != null && equipment.getTemperature() != null &&
230 equipment.getTemperature() > equipment.getTemperatureThreshold()) {
231 isError = true;
232 errorReason = "Temperature exceeded threshold";
233 }
234
235 // 检查压力阈值
236 if (!isError && equipment.getPressureThreshold() != null && equipment.getPressure() != null &&
237 equipment.getPressure() > equipment.getPressureThreshold()) {
238 isError = true;
239 errorReason = "Pressure exceeded threshold";
240 }
241
242 // 检查振动阈值
243 if (!isError && equipment.getVibrationThreshold() != null && equipment.getVibration() != null &&
244 equipment.getVibration() > equipment.getVibrationThreshold()) {
245 isError = true;
246 errorReason = "Vibration exceeded threshold";
247 }
248
249 // 如果有错误，则更新状态
250 if (isError && equipment.getStatus() != EquipmentMonitor.EquipmentStatus.ERROR) {
251 equipment.setStatus(EquipmentMonitor.EquipmentStatus.ERROR);
252 // 记录状态变更
253 recordStatusChange(equipment.getEquipmentCode(), EquipmentMonitor.EquipmentStatus.ERROR, "SYSTEM");
254 } else if (!isError && equipment.getStatus() == EquipmentMonitor.EquipmentStatus.ERROR &&
255 equipment.getStatus() != EquipmentMonitor.EquipmentStatus.MAINTENANCE) {
256 // 如果错误恢复且不是维护状态，则恢复到在线状态
257 equipment.setStatus(EquipmentMonitor.EquipmentStatus.ONLINE);
258 // 记录状态变更
259 recordStatusChange(equipment.getEquipmentCode(), EquipmentMonitor.EquipmentStatus.ONLINE, "SYSTEM");
260 }

270 private void recordStatusChange(String equipmentCode, EquipmentMonitor.EquipmentStatus status, String operator) {
271 EquipmentStatusRecord record = new EquipmentStatusRecord();
272 record.setEquipmentCode(equipmentCode);
273 record.setStatus(convertStatus(status));
274 record.setOperator(operator);
275
276 // 设置状态变更时的设备指标
277 Optional<EquipmentMonitor> optionalEquipment = equipmentMonitorRepository.findByEquipmentCode(equipmentCode);
278 if (optionalEquipment.isPresent()) {
279 EquipmentMonitor equipment = optionalEquipment.get();
280 record.setTemperature(equipment.getTemperature());
281 record.setPressure(equipment.getPressure());
282 record.setHumidity(equipment.getHumidity());
283 // 进行类型转换：Double -> Integer
284 if (equipment.getVibration() != null) {
285 record.setVibration(equipment.getVibration().intValue());
286 }
287 }
288
289 equipmentStatusRecordRepository.save(record);
290 }
