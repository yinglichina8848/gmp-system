102 .header(HttpHeaders.CONTENT_LENGTH, String.valueOf(fileContent.length))
103 .contentType(MediaType.APPLICATION_OCTET_STREAM)
104 .body(resource);

113 private String getFileNameFromPath(String filePath) {
114 if (filePath == null || filePath.isEmpty()) {
115 return "downloaded_file";
116 }
117 int lastSeparatorIndex = Math.max(
118 filePath.lastIndexOf('/'),
119 filePath.lastIndexOf('\\'));
120 if (lastSeparatorIndex != -1) {
121 return filePath.substring(lastSeparatorIndex + 1);
122 }
123 return filePath;
124 }

301 private Object getFieldValue(Object object, String fieldName, Object defaultValue) throws Exception {
302 Field field = findField(object.getClass(), fieldName);
303 if (field != null) {
304 boolean accessible = field.isAccessible();
305 field.setAccessible(true);
306 try {
307 return field.get(object);
308 } finally {
309 field.setAccessible(accessible);
310 }
311 }
312 return defaultValue;
313 }
314
318 private Field findField(Class<?> clazz, String fieldName) {
319 for (Field field : clazz.getDeclaredFields()) {
320 if (field.getName().equals(fieldName)) {
321 return field;
322 }
323 }
324 Class<?> superClass = clazz.getSuperclass();
325 if (superClass != null && superClass != Object.class) {
326 return findField(superClass, fieldName);
327 }
328 return null;
329 }
