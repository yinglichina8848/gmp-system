294 @Override
295 public String generateNewVersionNumber(Long documentId, String versionType) {
296 // 获取最新版本号
297 Optional<String> latestVersionOpt = documentVersionRepository.findLatestVersionNumberByDocumentId(documentId);
298
299 if (latestVersionOpt.isPresent()) {
300 // 解析版本号并递增
301 String latestVersion = latestVersionOpt.get();
302 try {
303 // 假设版本号格式为 "1.0.0"
304 String[] parts = latestVersion.split("\\.");
305 if (parts.length == 3) {
306 int major = Integer.parseInt(parts[0]);
307 int minor = Integer.parseInt(parts[1]);
308 int patch = Integer.parseInt(parts[2]);
309
310 // 根据版本类型递增不同部分
311 if ("MAJOR".equals(versionType)) {
312 major += 1;
313 minor = 0;
314 patch = 0;
315 } else if ("MINOR".equals(versionType)) {
316 minor += 1;
317 patch = 0;
318 } else {
319 patch += 1;
320 }
321
322 return major + "." + minor + "." + patch;
323 }
324 } catch (NumberFormatException e) {
325 // 版本号格式不正确，重新开始
326 }
327 }
328
329 // 如果没有版本或格式不正确，从 1.0.0 开始
330 return "1.0.0";
331 }

418 protected String calculateChecksum(MultipartFile file) throws IOException {
419 try {
420 MessageDigest md = MessageDigest.getInstance("SHA-256");
421 try (InputStream inputStream = file.getInputStream()) {
422 byte[] buffer = new byte[8192];
423 int bytesRead;
424 while ((bytesRead = inputStream.read(buffer)) != -1) {
425 md.update(buffer, 0, bytesRead);
426 }
427 }
428
429 byte[] hashBytes = md.digest();
430 StringBuilder sb = new StringBuilder();
431 for (byte b : hashBytes) {
432 sb.append(String.format("%02x", b));
433 }
434 return sb.toString();
435 } catch (NoSuchAlgorithmException e) {
436 throw new IOException("SHA-256算法不可用", e);
437 }
438 }

517 private Object getFieldValue(Object obj, String fieldName) {
518 if (obj == null || fieldName == null) {
519 return null;
520 }
521
522 try {
523 java.lang.reflect.Field field = findField(obj.getClass(), fieldName);
524 if (field != null) {
525 field.setAccessible(true);
526 return field.get(obj);
527 }
528 // 尝试通过getter方法获取
529 String getterName = "get" + Character.toUpperCase(fieldName.charAt(0)) + fieldName.substring(1);
530 try {
531 java.lang.reflect.Method getter = obj.getClass().getMethod(getterName);
532 return getter.invoke(obj);
533 } catch (NoSuchMethodException e) {
534 // 如果getter不存在，返回null
535 return null;
536 }
537 } catch (Exception e) {
538 // 反射访问失败，返回null
539 return null;
540 }

546 private void setFieldValue(Object obj, String fieldName, Object value) {
547 if (obj == null || fieldName == null) {
548 return;
549 }
550
551 try {
552 java.lang.reflect.Field field = findField(obj.getClass(), fieldName);
553 if (field != null) {
554 field.setAccessible(true);
555 field.set(obj, value);
556 } else {
557 // 尝试通过setter方法设置
558 String setterName = "set" + Character.toUpperCase(fieldName.charAt(0)) + fieldName.substring(1);
559 try {
560 Class<?> fieldType = value != null ? value.getClass() : Object.class;
561 java.lang.reflect.Method setter = obj.getClass().getMethod(setterName, fieldType);
562 setter.invoke(obj, value);
563 } catch (NoSuchMethodException e) {
564 // 如果setter不存在，尝试查找匹配参数类型的setter
565 java.lang.reflect.Method[] methods = obj.getClass().getMethods();
566 for (java.lang.reflect.Method method : methods) {
567 if (method.getName().equals(setterName) && method.getParameterCount() == 1) {
568 method.invoke(obj, value);
569 break;
570 }
571 }
572 }
573 }
574 } catch (Exception e) {
575 // 反射设置失败，静默忽略
576 }
577 }

582 private java.lang.reflect.Field findField(Class<?> clazz, String fieldName) {
583 Class<?> currentClass = clazz;
584 while (currentClass != null && currentClass != Object.class) {
585 try {
586 return currentClass.getDeclaredField(fieldName);
587 } catch (NoSuchFieldException e) {
588 currentClass = currentClass.getSuperclass();
589 }
590 }
591 return null;
592 }
