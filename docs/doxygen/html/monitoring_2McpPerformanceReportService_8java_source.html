111 private String getTimeAgo(int value, String unit) {
112 Calendar calendar = Calendar.getInstance();
113 switch (unit) {
114 case "minute":
115 calendar.add(Calendar.MINUTE, -value);
116 break;
117 case "hour":
118 calendar.add(Calendar.HOUR_OF_DAY, -value);
119 break;
120 case "day":
121 calendar.add(Calendar.DAY_OF_MONTH, -value);
122 break;
123 case "week":
124 calendar.add(Calendar.WEEK_OF_YEAR, -value);
125 break;
126 }
127 return calendar.getTime().toString();
