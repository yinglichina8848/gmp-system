42 @Override
43 @Transactional
44 public SalesOrderDTO createSalesOrder(SalesOrderRequestDTO request, String createdBy) {
45 // 验证客户是否存在
46 Customer customer = customerRepository.findById(request.getCustomerId())
47 .orElseThrow(() -> new IllegalArgumentException("客户不存在: " + request.getCustomerId()));
48
49 // 创建销售订单
50 SalesOrder salesOrder = new SalesOrder();
51 salesOrder.setCustomer(customer);
52 salesOrder.setOrderNumber(generateOrderNumber());
53 salesOrder.setStatus("PENDING");
54 salesOrder.setRemarks(request.getRemarks());
55 salesOrder.setCreatedBy(createdBy);
56
57 // 添加订单项
58 for (SalesOrderRequestDTO.OrderItemRequestDTO itemRequest : request.getItems()) {
59 OrderItem orderItem = new OrderItem();
60 orderItem.setProductCode(itemRequest.getProductCode());
61 orderItem.setProductName(itemRequest.getProductName());
62 orderItem.setUnitPrice(new BigDecimal(itemRequest.getUnitPrice()));
63 orderItem.setQuantity(itemRequest.getQuantity());
64 orderItem.setRemarks(itemRequest.getRemarks());
65
66 // 计算小计
67 BigDecimal quantity = new BigDecimal(orderItem.getQuantity());
68 orderItem.setSubTotal(orderItem.getUnitPrice().multiply(quantity));
69
70 // 添加到订单
71 salesOrder.addItem(orderItem);
72 }
73
74 // 保存订单
75 SalesOrder savedOrder = salesOrderRepository.save(salesOrder);
76
77 // 转换为DTO
78 return modelMapper.map(savedOrder, SalesOrderDTO.class);
79 }
