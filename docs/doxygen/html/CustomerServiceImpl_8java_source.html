34 @Override

36 public CustomerDTO createCustomer(CustomerRequestDTO request) {
37 // 检查客户代码是否已存在
38 if (customerRepository.findByCode(request.getCode()).isPresent()) {
39 throw new IllegalArgumentException("客户代码已存在: " + request.getCode());
40 }
41
42 // 转换请求DTO为实体
43 Customer customer = modelMapper.map(request, Customer.class);
44 customer.setStatus("ACTIVE"); // 默认激活状态
45
46 // 保存客户
47 Customer savedCustomer = customerRepository.save(customer);

49 // 转换为响应DTO

53 @Override
54 @Transactional
55 public CustomerDTO updateCustomer(Long id, CustomerRequestDTO request) {
56 // 查找客户
57 Customer customer = customerRepository.findById(id)
58 .orElseThrow(() -> new IllegalArgumentException("客户不存在: " + id));
59
60 // 检查客户代码是否已被其他客户使用
61 if (request.getCode() != null && !request.getCode().equals(customer.getCode()) &&
62 customerRepository.existsByCodeAndIdNot(request.getCode(), id)) {
63 throw new IllegalArgumentException("客户代码已存在: " + request.getCode());
64 }
65
66 // 更新客户信息
67 modelMapper.map(request, customer);
68
69 // 保存更新
70 Customer updatedCustomer = customerRepository.save(customer);

72 // 转换为响应DTO
