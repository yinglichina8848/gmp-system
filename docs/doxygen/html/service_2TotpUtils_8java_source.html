45 public String generateTotpCode(String secretKey, long timestamp) {
46 try {
47 if (secretKey == null || secretKey.isEmpty()) {
48 return null;
49 }
50 byte[] key = Base64.getDecoder().decode(secretKey);
51 long timeStep = timestamp / TOTP_TIME_STEP;
52
53 // 将时间戳转换为字节数组
54 ByteBuffer buffer = ByteBuffer.allocate(8);
55 buffer.putLong(0, timeStep);
56 byte[] timeBytes = buffer.array();
57
58 // 使用HMAC-SHA1算法计算
59 byte[] hmacResult = hmacSha1(key, timeBytes);
60
61 // 动态截断
62 int offset = hmacResult[hmacResult.length - 1] & 0xF;
63 int binary = ((hmacResult[offset] & 0x7F) << 24) |
64 ((hmacResult[offset + 1] & 0xFF) << 16) |
65 ((hmacResult[offset + 2] & 0xFF) << 8) |
66 (hmacResult[offset + 3] & 0xFF);
67
68 // 生成6位验证码
69 int otp = binary % (int) Math.pow(10, TOTP_CODE_LENGTH);
70 return String.format("%0" + TOTP_CODE_LENGTH + "d", otp);
71 } catch (Exception e) {
72 return null;
