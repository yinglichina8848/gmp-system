104 try {
105 // 确保上传目录存在

108
109 // 生成唯一文件名
110 String originalFilename = file.getOriginalFilename();
111 String extension = originalFilename != null ? originalFilename.substring(originalFilename.lastIndexOf('.'))
112 : "";
113 String fileName = UUID.randomUUID() + extension;
114
115 // 保存文件
116 Path filePath = uploadPath.resolve(fileName);
117 Files.copy(file.getInputStream(), filePath);
118
119 // 创建附件记录
120 DeviationAttachment attachment = new DeviationAttachment();
121 // 使用setter方法设置属性值
122 attachment.setDeviation(deviation);
123 attachment.setFileName(originalFilename);
124 attachment.setFilePath(filePath.toString());
125 attachment.setFileSize(file.getSize());
126 attachment.setFileType(file.getContentType());
127 attachment.setDescription(description);
128 // 设置默认上传人为系统用户ID 1
129 attachment.setUploadedBy(1L); // 使用Long类型的默认用户ID
130 // 添加上传时间
131 attachment.setUploadedAt(LocalDateTime.now());
132
133 // 直接保存附件，而不是通过deviation的关系
134 deviationAttachmentRepository.save(attachment);
135 return deviation;
136 } catch (IOException e) {
