197 private Object getFieldValue(Object obj, String fieldName) {
198 if (obj == null || fieldName == null) {
199 return null;
200 }
201
202 try {
203 // 尝试通过字段访问
204 Field field = findField(obj.getClass(), fieldName);
205 if (field != null) {
206 field.setAccessible(true);
207 return field.get(obj);
208 }
209 // 尝试通过getter方法获取
210 String getterName = "get" + Character.toUpperCase(fieldName.charAt(0)) + fieldName.substring(1);
211 try {
212 Method getter = obj.getClass().getMethod(getterName);
213 return getter.invoke(obj);
214 } catch (NoSuchMethodException e) {
215 // 如果getter不存在，返回null
216 return null;
217 }
218 } catch (Exception e) {
219 log.error("获取字段值失败: {}.{}", obj.getClass().getSimpleName(), fieldName, e);
220 return null;
221 }
222 }

227 private Field findField(Class<?> clazz, String fieldName) {
228 Class<?> currentClass = clazz;
229 while (currentClass != null && currentClass != Object.class) {
230 try {
231 return currentClass.getDeclaredField(fieldName);
232 } catch (NoSuchFieldException e) {
233 currentClass = currentClass.getSuperclass();
234 }
235 }
236 return null;
237 }

267 private boolean evaluatePermissions(Set<String> roles, List<PermissionPolicy> policies,
268 PermissionType required, PermissionContext context) {
269 // 支持复杂的权限规则：
270 // - 基于角色的访问控制
271 // - 基于属性的访问控制
272 // - 时间窗口限制
273 // - 地理位置限制
274
275 // 1. 检查管理员权限
276 if (isAdmin(roles)) {
277 return true;
278 }
279
280 // 2. 检查文档所有者权限
281 if (context.getUserId().equals(getDocumentAuthor(context.getDocumentId()))) {
282 return true;
283 }
284
285 // 3. 评估策略权限
286 return policies.stream()
287 .anyMatch(policy -> policy.evaluate(roles, required, context));
288 }
