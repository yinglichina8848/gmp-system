81 public String generateToken(String username, Collection<String> roles, List<String> permissions) {
82 Date now = new Date();
83 Date expiryDate = new Date(now.getTime() + expiration);
84
85 Map<String, Object> claims = new HashMap<>();
86 claims.put("sub", username);
87 claims.put("iat", now.getTime() / 1000);
88 claims.put("exp", expiryDate.getTime() / 1000);
89 claims.put("iss", issuer);
90 claims.put("roles", roles);
91 claims.put("permissions", permissions);
92
93 return Jwts.builder()
94 .setClaims(claims)
95 .setIssuedAt(now)
96 .setExpiration(expiryDate)
97 .signWith(getSecretKey(), SignatureAlgorithm.HS256)
98 .compact();
