39 @PostMapping("/files")
40 @Operation(summary = "兼容原File服务 - 上传文件")
41 public ResponseEntity<?> uploadFile(
42 @RequestParam("file") MultipartFile file,
43 @RequestParam("type") String type,
44 @RequestParam(required = false) Map<String, Object> metadata) {
45 try {
46 // 使用type参数作为module值
47 CommonFileDTO fileDTO = commonFileService.uploadFile(file, type, metadata);
48
49 // 返回Map以避免类型兼容性问题
50 Map<String, Object> response = new HashMap<>();
51 response.put("id", fileDTO.getId());
52 response.put("fileName", fileDTO.getFileName() != null ? fileDTO.getFileName() : file.getOriginalFilename());
53 response.put("fileType", fileDTO.getFileType() != null ? fileDTO.getFileType() : file.getContentType());
54 response.put("fileSize", fileDTO.getFileSize() != null ? fileDTO.getFileSize() : file.getSize());
55 response.put("uploadTime", System.currentTimeMillis());
56
57 return ResponseEntity.status(HttpStatus.CREATED).body(response);
58 } catch (Exception e) {
59 Map<String, String> errorResponse = new HashMap<>();
60 errorResponse.put("error", "文件上传失败: " + e.getMessage());
61 return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR).body(errorResponse);
62 }
63 }

68 @GetMapping("/files/{fileId}/download")
69 @Operation(summary = "兼容原File服务 - 下载文件")
70 public ResponseEntity<byte[]> downloadFile(@PathVariable Long fileId) {
71 try {
72 CommonFileDTO fileInfo = commonFileService.getFileInfo(fileId);
73 InputStream inputStream = commonFileService.downloadFile(fileId);
74 byte[] bytes = inputStream.readAllBytes();
75
76 HttpHeaders headers = new HttpHeaders();
77 String fileName = fileInfo.getFileName() != null ? fileInfo.getFileName() : "file.bin";
78 String fileType = fileInfo.getFileType() != null ? fileInfo.getFileType() : "application/octet-stream";
79 headers.setContentType(MediaType.parseMediaType(fileType));
80 headers.setContentDispositionFormData("attachment", fileName);
81 headers.setContentLength(bytes.length);
82
83 return ResponseEntity.ok().headers(headers).body(bytes);
84 } catch (Exception e) {
85 return ResponseEntity.status(HttpStatus.NOT_FOUND).body(null);
86 }
