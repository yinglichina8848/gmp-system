58 public String validatePasswordComplexity(String password) {
59 if (password == null || password.length() < minPasswordLength) {
60 return "密码长度必须至少为" + minPasswordLength + "个字符";
61 }
62
63 if (!HAS_DIGIT.matcher(password).find()) {
