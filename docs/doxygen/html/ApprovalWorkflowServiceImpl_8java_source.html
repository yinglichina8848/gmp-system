84 @Override
85 public boolean executeApprovalStep(Long instanceId, String userId, String decision, String comments) {
86 try {
87 // 查找审批实例
88 ApprovalInstance instance = instanceRepository.findById(instanceId)
89 .orElseThrow(() -> new RuntimeException("审批实例不存在: " + instanceId));
90
91 // 验证审批人权限
92 if (!validateApproverPermission(instance, userId)) {
93 throw new RuntimeException("审批人无权限审批此任务");
94 }
95
96 // 记录审批决定
97 recordApprovalDecision(instance, userId, decision, comments);
98
99 // 计算下一步
100 String workflowDefinition = (String) getFieldValue(instance, "workflowDefinition");
101 JsonNode workflowDef = objectMapper.readTree(workflowDefinition);
102 String currentStep = (String) getFieldValue(instance, "currentStep");
103 String nextStep = calculateNextStep(workflowDef, currentStep, decision);
104
105 if (nextStep != null) {
106 // 分配下一步审批任务
107 assignNextStep(instance, workflowDef, nextStep);
108 setFieldValue(instance, "currentStep", nextStep);
109 instanceRepository.save(instance);
110 } else {
111 // 结束审批流程
112 finalizeApprovalProcess(instance, decision);
113 }
114
115 // 发送通知
116 notifyApprovers(instance, workflowDef);
117
118 return true;
119 } catch (Exception e) {
120 log.error("执行审批步骤失败: {}", e.getMessage(), e);
121 return false;
122 }

285 private String calculateNextStep(JsonNode workflowDef, String currentStep, String decision) {
286 // 根据当前步骤和决策计算下一步
287 if (!workflowDef.has("steps") || !workflowDef.get("steps").isArray()) {
288 return null;
289 }
290
291 JsonNode steps = workflowDef.get("steps");
292 for (int i = 0; i < steps.size(); i++) {
293 JsonNode step = steps.get(i);
294 if (step.has("name") && currentStep.equals(step.get("name").asText())) {
295 // 返回下一步骤
296 if (i + 1 < steps.size()) {
297 return steps.get(i + 1).get("name").asText();
298 }
299 return null; // 没有下一步骤，流程结束
300 }
301 }
302 return null;
303 }
