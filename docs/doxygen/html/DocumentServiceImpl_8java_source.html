171 private String calculateChecksum(MultipartFile file) throws IOException, NoSuchAlgorithmException {
172 MessageDigest md = MessageDigest.getInstance("SHA-256");
173 try (InputStream inputStream = file.getInputStream()) {
174 byte[] buffer = new byte[8192];
175 int read;
176 while ((read = inputStream.read(buffer)) > 0) {
177 md.update(buffer, 0, read);
178 }
179 }
180
181 byte[] hashBytes = md.digest();
182 StringBuilder hexString = new StringBuilder();
183 for (byte hashByte : hashBytes) {
184 String hex = Integer.toHexString(0xff & hashByte);
185 if (hex.length() == 1) {
186 hexString.append('0');
187 }
188 hexString.append(hex);
189 }
190 return hexString.toString();
191 }

445 protected String generateDocCode() {
446 // 生成格式为 DOC_YYYYMMDD_0001 的文档编号
447 String dateStr = LocalDateTime.now().format(java.time.format.DateTimeFormatter.ofPattern("yyyyMMdd"));
448 String prefix = "DOC_" + dateStr + "_";
449
450 // 查找当天最后一个文档编号
451 List<Document> documents = documentRepository.findAll();
452 long maxNum = documents.stream()
453 .filter(doc -> {
454 try {
455 String docNum = (String) getFieldValue(doc, "documentNumber");
456 return docNum != null && docNum.startsWith(prefix);
457 } catch (Exception e) {
458 return false;
459 }
460 })
461 .mapToLong(doc -> {
462 try {
463 String docNum = (String) getFieldValue(doc, "documentNumber");
464 String numStr = docNum.substring(prefix.length());
465 return Long.parseLong(numStr);
466 } catch (Exception e) {
467 return 0;
468 }
469 })
470 .max()
471 .orElse(0);
472
473 // 生成新编号
474 return prefix + String.format("%04d", maxNum + 1);
475 }
