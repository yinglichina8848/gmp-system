122 private String extractFileNameFromPath(String filePath) {
123 if (filePath == null || filePath.isEmpty()) {
124 return "unknown_file";
125 }
126 int lastSeparatorIndex = Math.max(
127 filePath.lastIndexOf('/'),
128 filePath.lastIndexOf('\\'));
129 if (lastSeparatorIndex != -1) {
130 return filePath.substring(lastSeparatorIndex + 1);
131 }
132 return filePath;
133 }

135 private String formatFileSize(long size) {
136 if (size <= 0) return "0 B";
137 final String[] units = new String[]{"B", "KB", "MB", "GB", "TB"};
138 int digitGroups = (int) (Math.log10(size) / Math.log10(1024));
139 return String.format("%.2f %s", size / Math.pow(1024, digitGroups), units[digitGroups]);
140 }
141
142 private String detectFileType(String fileName) {
143 if (fileName == null || !fileName.contains(".")) {
144 return "unknown";
145 }
146 return fileName.substring(fileName.lastIndexOf(".") + 1).toLowerCase();
147 }
