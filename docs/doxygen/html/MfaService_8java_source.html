101 int offset = hmacResult[hmacResult.length - 1] & 0xF;
102 int binary = ((hmacResult[offset] & 0x7F) << 24) |
103 ((hmacResult[offset + 1] & 0xFF) << 16) |
104 ((hmacResult[offset + 2] & 0xFF) << 8) |
105 (hmacResult[offset + 3] & 0xFF);
106
107 // 生成6位验证码
108 int otp = binary % (int) Math.pow(10, TOTP_CODE_LENGTH);
109 return String.format("%0" + TOTP_CODE_LENGTH + "d", otp);
110 } catch (Exception e) {

112 }
